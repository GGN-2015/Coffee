IMPORT "import/io.txt"


FUNC isprime(x)
    IF x <= 1 THEN        ; x is not prime
        RETURN 0
    ENDIF
    FOR i := 2 TO x-1 DO
        IF x % i = 0 THEN ; x is not prime
            RETURN 0
        ENDIF
    ENDFOR
    RETURN 1
ENDFUNC


FUNC main()
    FOR i := 1 TO 1000 DO
        IF isprime(i) THEN
            CALL printInt(i)
            PUTCHAR(' ')
        ENDIF
    ENDFOR
ENDFUNC
